#include <stdio.h>
int main()
{
	printf("Hello, World!"); /*printf() outputs the quoted string*/
	if (n == 0 || n == 1){
		return n;
	}
	j = 0;
	for (i = 0; i < n-1; i++){
		if (arr[i] != arr[i+1]){
			arr[j] = arr[i];
			j++;
		}
	}
	arr[j++] = arr[n-1];
	return 0;
}
